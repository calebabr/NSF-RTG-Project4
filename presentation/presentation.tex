\begin{frame}{Motivation}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \textbf{The overparameterization puzzle:}
      \begin{itemize}
        \item Networks have far more parameters than the problem demands
        \item Classical theory predicts overfitting;
              practice shows excellent generalization
        \item \alert{Why?}
      \end{itemize}
      \vspace{0.6em}
      \textbf{Bias collapse gives a concrete mechanism:}
      \begin{itemize}
        \item Gradient descent discovers that most neurons are
              redundant and \emph{eliminates them automatically}
        \item $m = 1500$ neurons $\;\longrightarrow\;$
              effective complexity of $k = 7$ neurons
        \item No pruning, no regularization — the optimizer acts alone
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{tcolorbox}[slideboxA, title={This project}]
        \small
        Build rigorous numerical evidence for:
        \vspace{4pt}
        \begin{enumerate}
          \item \textbf{OP 4.1:} $C(m,\fstar) \to k$ as $m\to\infty$,\\
                where $k$ = inflection-point count of $\fstar$
          \vspace{4pt}
          \item \textbf{OP 4.3:} After collapse, the network
                can be pruned to $k$ neurons with a
                provable $L^2$ error bound
        \end{enumerate}
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

%% ============================================================
\begin{frame}{Setup}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Model:} 1D shallow ReLU network
      \[
        f(x) = \sum_{j=1}^{m} a_j\,\sigma(x - b_j),
        \quad \sigma(z)=\max(0,z), \quad x\!\in\![-1,1]
      \]
      Each neuron places a \emph{kink} at $b_j$
      with slope change $a_j$.

      \vspace{0.5em}
      \textbf{Training:} Gradient flow on continuous MSE
      \[
        \mathcal{L}
        = \tfrac{1}{2}\int_{-1}^{1}(f - \fstar)^2\,dx
      \]
      ODEs $\;\dot{a}_j = -\partial\mathcal{L}/\partial a_j$,\;
      $\dot{b}_j = -\partial\mathcal{L}/\partial b_j$\;
      solved numerically from $t = 0$ to $t = T$.
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Open Problem 4.1}
        \small
        \[
          \lim_{m\to\infty} C(m,\fstar)
          \;=\; k
          \;:=\;
          \abs{\set{x \in [-1,1] : (\fstar)''(x)=0,\;\uparrow\!\downarrow}}
        \]
        \vspace{-0.3em}
        \begin{enumerate}\small
          \item[\textbf{4.1.1}] Prove $C \leq k$ independent of $m$
          \item[\textbf{4.1.2}] Finite-time vs.\ asymptotic convergence
          \item[\textbf{4.1.3}] Sign pattern vs.\ curvature amplitude
        \end{enumerate}
      \end{block}
      \vspace{0.35em}
      \begin{block}{Open Problem 4.3}
        \small
        \[
          \norm{\ftilde - f}_{L^2}
          \;\leq\; \delta \cdot \sum_{j=1}^{m} \abs{a_j}
        \]
        $\delta$ = max intra-cluster diameter;
        $\ftilde$ = pruned $k$-neuron network.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Contributions}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Numerical evidence for OP 4.1:}
      \begin{itemize}
        \item[\color{darkgreen}\checkmark]
              $C(m,\fstar)=k$ confirmed exactly:
              $\sin(\pi x)$ at $m\!\geq\!1000$;
              $\sin(4\pi x)$ at $m\!=\!1500$
        \item[\color{darkgreen}\checkmark]
              Non-monotone collapse dynamics characterized
        \item[\color{darkgreen}\checkmark]
              ODE stationarity verified:
              $|\dot{a}_j|,|\dot{b}_j|\approx 0$
              and $R_j\approx 0$ at convergence
        \item[\color{darkgreen}\checkmark]
              Instability framework built for
              $|C-k|\leq 2$ (3 categories, 37 tests)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Numerical evidence for OP 4.3:}
      \begin{itemize}
        \item[\color{darkgreen}\checkmark]
              Pruning bound holds for \textbf{108\,/\,108} runs
        \item[\color{darkgreen}\checkmark]
              Tightness $\in[0.001,\,0.15]$:
              bound satisfied, not saturated
        \item[\color{auorange}$\circ$]
              $\sum_j|a_j|$ grows with $m$ —
              bounding it is the key obstacle
              to a formal proof
      \end{itemize}
      \vspace{0.5em}
      \begin{tcolorbox}[slideboxA, title={Ongoing}]
        \small

        Local parallel: $m$ up to 1000, $T\!\leq\!10000$.\\

\begin{frame}{Open Problems and Research Directions}
  \begin{columns}[T]
    \begin{column}{0.49\textwidth}
      \small
      \textbf{OP 4.1 — remaining goals:}
      \begin{itemize}
        \item \textbf{4.1.1:} Formal proof $C \leq k$.
              Instability evidence points toward a
              Lyapunov argument: any $C>k$ state is
              linearly unstable $\Rightarrow$ $C$
              cannot remain above $k$.
        \item \textbf{4.1.2:} Finite-time vs.\
              asymptotic. The $T\in\{200,\ldots,10000\}$
              progression tests whether $C$ stabilizes
              at a finite $T^\ast$.
        \item \textbf{4.1.3:} Sign pattern vs.\
              amplitude: compare $\sin(2\pi x)$ with
              $c\sin(2\pi x)$ for varying $c$.
      \end{itemize}
    \end{column}
    \begin{column}{0.49\textwidth}
      \small
      \textbf{OP 4.3 — remaining challenge:}
      \begin{itemize}
        \item Elementary estimate
              $\norm{\ftilde-f}\leq\delta\cdot\sum_j|a_j|$
              follows from Lipschitz-$\sigma$.
        \item Hard part: bounding $\sum_j|a_j|$.
              If $\sum_j|a_j|=O(m^\alpha)$ and
              $\delta=O(m^{-\beta})$ with $\beta>\alpha$,
              the bound is non-vacuous.
        \item Data suggests sublinear growth;
              quantifying $\alpha$ is the next step.
      \end{itemize}
      \vspace{0.4em}
      \begin{tcolorbox}[slideboxA, title={OP 4.2 (future)}]
        \small
        Higher dimensions $d\geq 2$: does collapse
        generalize to hyperplane configurations?
        Natural extension once 1D is settled.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

        \small
        Gradient descent is not only an optimizer —
        it is a \textbf{compression algorithm} that
        finds the minimum-complexity solution
        matching the target's intrinsic structure.

        \vspace{0.4em}
        Proving this rigorously provides a
        mechanistic explanation for generalization
        in overparameterized networks — one of the
