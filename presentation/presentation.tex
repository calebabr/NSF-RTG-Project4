\begin{frame}{Motivation}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \textbf{The overparameterization puzzle:}
      \begin{itemize}\small
        \item Modern networks have far more parameters than
              the problem demands
        \item Classical learning theory predicts overfitting;
              practice shows robust generalization
        \item \alert{Why does the optimizer self-simplify?}
      \end{itemize}
      \vspace{0.4em}
      \textbf{Bias collapse: a concrete mechanism}
      \begin{itemize}\small
        \item Gradient flow automatically eliminates redundant
              neurons, with no regularization required
        \item $m = 1500$ neurons collapses to $\sim7$ distinct values.
        \item The number of surviving \emph{bias clusters}
              is approximately around the number of \emph{inflection points}
              of the target function
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{tcolorbox}[slideboxA, title={This work}]
        \small
        Rigorous numerical investigation of three
        open problems:
        \vspace{4pt}
        \begin{enumerate}
          \item \textbf{OP 4.1:} Does $C(m,\fstar) \to k$
                as $m \to \infty$?
          \item \textbf{OP 4.2:} Does collapse generalize
                to $d \geq 2$ dimensions?
          \item \textbf{OP 4.3:} Is there a provable
                $L^2$ pruning bound after collapse?
        \end{enumerate}
        \vspace{4pt}
        \textbf{72 runs,} 9 targets, $m$ up to 5000.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

%% ============================================================
\begin{frame}{Model and Gradient Flow}
  \begin{columns}[T]
    \begin{column}{0.57\textwidth}
      \textbf{1D shallow ReLU network:}
      \[
        f(x) = \sum_{j=1}^{m} a_j\,\sigma(x - b_j),
        \quad \sigma = \max(0,\cdot),\quad x \in [-1,1]
      \]
      Each neuron places a \emph{kink} at bias $b_j$
      with slope change $a_j$.

      \vspace{0.3em}
      \textbf{Gradient flow on MSE\;} $\mathcal{L}
        = \tfrac{1}{2}\int_{-1}^{1}(f - \fstar)^2\,dx$\textbf{:}
      \begin{align*}
        \dot{a}_j &= -\!\int_{-1}^{1}\!(f-\fstar)\,\sigma(x-b_j)\,dx \\[-2pt]
        \dot{b}_j &= a_j\!\int_{b_j}^{1}\!(f-\fstar)\,dx
      \end{align*}
      \vspace{2pt}
      \small Solved via \texttt{scipy} RK45, $T=500$,
      rtol\,$=10^{-4}$. Stationarity criterion:
      $\max_j|\dot{a}_j|,|\dot{b}_j| < 0.01$.
    \end{column}
    \begin{column}{0.41\textwidth}
      \textbf{Key measured quantity:}
      \[
        C(m,\fstar) \;=\; \#\{\text{bias clusters}\}
      \]
      \small at stationarity; gap threshold 0.02.

      \vspace{0.2em}
      \begin{tcolorbox}[slideboxB, title={Fixed-point condition}]
        \small
        Each \emph{active} neuron ($a_j \neq 0$) satisfies
        $R_j \!:=\! \int_{b_j}^{1}\!(f{-}\fstar)\,dx \approx 0$
        at stationarity.
        Verified numerically at every converged run;
        connects to the theory of OP 4.1.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The Three Open Problems}
  \begin{columns}[T]
    \begin{column}{0.33\textwidth}
      \begin{block}{OP 4.1:Cluster Count}
        \small
        \[
          \lim_{m \to \infty} C(m,\fstar) \;=\; k
        \]
        Does the number of bias clusters converge
        to the number of sign-changing inflection
        points of $\fstar$?
      \end{block}
    \end{column}
    \begin{column}{0.33\textwidth}
      \begin{block}{OP 4.2:Higher Dimensions}
        \small
        For $\fstar : [-1,1]^2 \to \mathbb{R}$,
        does gradient flow concentrate the neuron
        measure onto a lower-dimensional set in
        hyperplane parameter space?
      \end{block}
    \end{column}
    \begin{column}{0.33\textwidth}
      \begin{block}{OP 4.3:Pruning Bound}
        \small
        After collapse, is there a provable $L^2$
        error bound when pruning to $k$ neurons?
        \[
          \norm{\ftilde - f}_{L^2}
          \leq \delta \cdot \sum_{j} \abs{a_j}
        \]
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.6em}
  \begin{center}
    \small
    Each open problem is introduced in detail immediately before its results.
  \end{center}
\end{frame}

%% ============================================================
%%  OP 4.1 — INTRO SLIDE (stated in full before results)
%% ============================================================
\begin{frame}{Open Problem 4.1: Cluster Count Conjecture}
  \begin{block}{Conjecture}
    \[
      \lim_{m \to \infty} C(m,\fstar) \;=\; k
    \]
    \vspace{-4pt}
    \small
    where $k :=$ \# sign-changing inflection points of $\fstar$ in $(-1,1)$
  \end{block}
  \vspace{0.6em}
  \small
  $C(m,\fstar)$ = number of distinct bias clusters at stationarity.\\[4pt]
  $k$ = number of sign-changing inflection points of the \textbf{target} $\fstar$.\\[4pt]
  $\fstar$ = the target function.

  \vspace{0.6em}
  \textbf{Sub-problems we address:}
  \begin{itemize}
    \item Can we prove $C(m,\fstar) \leq k$
          independently of $m$?
    \item Which neurons survive collapse
          to become cluster representatives?
  \end{itemize}
\end{frame}

\begin{frame}{Experimental Design}
  \begin{columns}[T]
    \begin{column}{0.40\textwidth}
      \textbf{Nine target functions:}\\[4pt]
      \renewcommand{\arraystretch}{1.1}
      \small
      \begin{tabular}{lcc}
        \toprule
        $\fstar(x)$ & $k$ & type \\
        \midrule
        $\sin(\pi x)$      & 1  & trig \\
        $x^3$              & 1  & poly \\
        $\sin(2\pi x)$     & 3  & trig \\
        $x^5 - 3x^3$       & 3  & poly \\
        $\sin(3\pi x)$     & 5  & trig \\
        $\sin(4\pi x)$     & 7  & trig \\
        $\sin(5\pi x)$     & 9  & trig \\
        $\sin(6\pi x)$     & 11 & trig \\
        $\sin(7\pi x)$     & 13 & trig \\
        \bottomrule
      \end{tabular}\\[4pt]
      \normalsize
      Trig: $k = 2n-1$.\quad
      Poly: irregular inflection spacing.
    \end{column}
    \begin{column}{0.58\textwidth}
      \small
      \textbf{Main sweep:}
      \begin{itemize}
        \item $m \in \{50,100,250,500,1000,2000,3500,5000\}$:
              number of neurons in the network
        \item $T = 500$: time horizon for ODE integration.
              SciPy's RK45 solver, a standard adaptive step-size
              integrator, numerically solves the gradient flow
              equations from $t=0$ to $t=T$, automatically
              adjusting step size to maintain accuracy.
        \item \textbf{72 completed runs} across 9 targets
      \end{itemize}
      \vspace{0.4em}
      \begin{tcolorbox}[slideboxA, title={Discrete GD comparison}]
        \small
        Fixed step-size discrete GD did not reproduce
        $C \to k$ for most targets. Adaptive step sizing
        (as in RK45) appears to matter for the clustering
        behavior to emerge.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[shrink=5]{OP 4.1: The Rise-Then-Fall Dynamic}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \textbf{Universal pattern across all 9 targets:}
      \vspace{4pt}
      \begin{enumerate}\small
        \item \textbf{Rise:} Small $m$ cannot populate all
              $k$ inflection attractors simultaneously.
              $C$ grows above $k$.
        \item \textbf{Peak:} $C$ reaches a maximum scaling
              with $k$ (up to $C = 40$ for $k = 11$).
        \item \textbf{Fall:} Collapse proceeds and $C$ decreases toward $k$.
      \end{enumerate}
      \vspace{0.4em}
      \begin{tcolorbox}[slideboxB, title={Key observation}]
        \small
        The peak of $C(m)$ shifts to \textbf{larger $m$}
        as $k$ grows. High-$k$ targets require much wider
        networks before the fall phase begins.
      \end{tcolorbox}
    \end{column}
    \begin{column}{0.52\textwidth}
      \scriptsize
      \textbf{Selected $C(m)$ at $T=500$:}\\[3pt]
      \renewcommand{\arraystretch}{1.15}
      \begin{tabular}{lc ccc}

\begin{frame}{OP 4.1 The Below-$k$ Phenomenon: Initialization Changes}
  \begin{columns}[T]

      \vspace{4pt}

      \scriptsize
      \renewcommand{\arraystretch}{1.2}
      \begin{tabular}{lcc ccc}
        \toprule
        Target & $k$ & $m$ & seed 42 & seed 5 & seed 25 \\
        \midrule

      \end{tabular}
      \vspace{4pt}

      \small
      $C$ \textbf{varies across seeds} in all three cases.
      No single $C < k$ value is consistent.
      \vspace{3pt}
      \begin{center}
        \convpanel{0}{2}\\[-2pt]
        {\tiny $\hookrightarrow$ $\sin(2\pi x)$, $k=3$: $C$ stays below $k$}
      \end{center}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{tcolorbox}[slideboxC, title={Interpretation}]
        \small
        $C$ varies across seeds $\Rightarrow$ $C < k$ states
        are \textbf{initialization-dependent local minima},
        not universal fixed points.

        \vspace{4pt}
        Gradient flow has multiple basins of attraction.
        The $k$-cluster basin exists but is not the only one.
      \end{tcolorbox}
      \vspace{0.3em}
      \begin{tcolorbox}[slideboxA, title={$x^5{-}3x^3$ seed 25: $C = k = 3$}]
        \small
        First time $x^5{-}3x^3$ reaches $C = k$.
        Shows the $k$-cluster attractor exists for
        polynomial targets, though harder to find.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[shrink=10]{Instability Near $k$ Results: Post Training Changes}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \small
      \textbf{Question:} Are states with $C \neq k$ unstable?

        why random initializations land at $C = k$.

\begin{frame}{Open Problem 4.2: Higher-Dimensional Collapse}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Question}
        For a structured target $\fstar : [-1,1]^2 \to \mathbb{R}$,
        does gradient flow concentrate the neuron measure onto
        a lower-dimensional set in hyperplane parameter space?
      \end{block}

      \vspace{0.3em}
      \small
      \textbf{2D network:}
      \[
        f(x) = \sum_{j=1}^{m} a_j\,\sigma(w_j^\top x + b_j),
        \quad x \in [-1,1]^2
      \]
      Each neuron defines an oriented hyperplane (line in 2D).
      Scale-invariant: $\theta_j = \angle(w_j)$,\; $\beta_j = b_j / \|w_j\|$.

      \vspace{0.2em}
      \textbf{Collapse in 2D:} the $(\theta_j, \beta_j)$ cloud
      concentrates onto a low-dimensional subset.
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{tcolorbox}[slideboxB, title={Diagnostic}]
        \small
        Angular entropy $H(\{\theta_j\})$: near 0 if collapsed;
        near $\log(36) \approx 3.58$ if uniform.
        \vspace{3pt}
        Effective neurons $N_{\rm eff} = e^H$:\\
        $N_{\rm eff}{=}1$ full collapse;\; $N_{\rm eff}{\approx}36$ uniform.
      \end{tcolorbox}
      \vspace{0.3em}
      \begin{tcolorbox}[slideboxA, title={Width sweep}]
        \small
        $m \in \{64, 256, 512, 1024\}$,\; 6000 epochs, Adam.
      \end{tcolorbox}
      \vspace{0.3em}
      \begin{tcolorbox}[slideboxA, title={Three targets studied}]
        \small
        Ridge, Separable, and Radial.\\
        See next slide for expected vs.\ observed behavior.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Open Problem 4.3: Pruning Bound}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Bound (from the slides)}
        \[
          \norm{\ftilde - f}_{L^2}
          \;\leq\;
          \delta \cdot \sum_{j=1}^{m} \abs{a_j}
        \]
      \end{block}

      \vspace{0.4em}
      \small
      \textbf{Setup:} After gradient flow, bias clusters
      have formed. \emph{Prune} by replacing each cluster
      with a single representative neuron, giving
      $\ftilde$ with only $k$ neurons.

      \vspace{0.3em}
      \textbf{$\delta$} = maximum intra-cluster diameter
      (how spread out the biases are within each cluster).

      \vspace{0.3em}
      \textbf{$\sum|a_j|$} = total weight mass of the
      full $m$-neuron network.

      \vspace{0.3em}
      \textbf{Why it matters:} Provides a formal $L^2$
      certificate that collapsing from $m$ neurons to
      $k$ neurons is safe, bridging the
      collapse phenomenon and approximation theory.
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{tcolorbox}[slideboxA, title={Why the bound holds}]
        \small
        Within a cluster, all $b_j$ differ by at most
        $\delta$, so $\sigma(x - b_j) \approx
        \sigma(x - b_{\rm center})$ up to an error $\leq \delta$.
        Summing over $m$ neurons, errors accumulate
        at most as $\delta \cdot \sum|a_j|$.
      \end{tcolorbox}
      \vspace{0.3em}
      \begin{tcolorbox}[slideboxB, title={The challenge}]
        \small
        For the bound to be \emph{non-vacuous}
        (goes to 0 as networks grow), we need
        $\delta \cdot \sum|a_j| \to 0$.

        \vspace{4pt}
        Our data shows $\sum|a_j| \sim \sqrt{m}$.
        So we need $\delta \to 0$ fast enough,
        which must come from longer training $T$,
        not wider networks $m$.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{OP 4.3: Pruning Bound Verification}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \includegraphics[width=\textwidth, height=0.72\textheight,
                       keepaspectratio]{pruning_bound_summary.png}
    \end{column}
    \begin{column}{0.42\textwidth}
      \small
      \textbf{Bound:}
      $\;\norm{\ftilde - f}_{L^2} \leq \delta \cdot \sum_j |a_j|$

      \vspace{0.4em}
      \begin{tcolorbox}[slideboxC, title={72\,/\,72 runs: bound holds}]
        \small
        Tightness (actual\,/\,bound):
        $0.0002$ to $0.148$.
        Substantial slack in all cases.
      \end{tcolorbox}

      \vspace{0.4em}
      \textbf{Key obstacle:}
      \begin{itemize}
        \item $\sum_j|a_j| \sim \sqrt{m}$ empirically
        \item $\delta$ does not shrink with $m$ after $m=500$
        \item Bound loosens $\sim\!\sqrt{m}$ while actual
              error plateaus
      \end{itemize}

      \vspace{0.35em}
      \begin{tcolorbox}[slideboxB, title={Path to non-vacuousness}]
        \small
        $\delta \to 0$ requires $T \to \infty$,
        \emph{not} $m \to \infty$.
        Study $\delta(T)$ at fixed $m$ to
        make the bound formally useful.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Lottery Ticket: Which Neurons Survive?}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \small
      \textbf{Setup:} At $t=0$, amplitudes
      $a_j \sim \mathcal{N}(0,0.01)$ are random.
      Only the bias placement $b_j$ is structured.
      Neurons near an inflection point are
      \emph{geometrically lucky}.

      \vspace{0.4em}
      \textbf{Five conditions per (target, $m$):}
      \texttt{full}, \texttt{geometric}, \texttt{bias\_only},
      \texttt{amp\_only}, \texttt{random\_k}\\[3pt]
      52 runs; 4 targets, $m \in \{500, 1000, 1500\}$.

      \vspace{0.4em}
      \begin{tcolorbox}[slideboxB, title={Claim 2 (performance): FALSE}]
        \small
        Lucky $k$ neurons alone achieve
        $100\times$--$50{,}000\times$ higher loss
        than the full network.
        $k$ kinks cannot approximate a smooth target;
        full redundancy is necessary.
      \end{tcolorbox}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{tcolorbox}[slideboxC,
                        title={Claim 1 (survival): bias position is key}]
        \small
        \textbf{geometric} $\approx$ \textbf{bias\_only}
        in every case (loss diff $< 0.002$).
        Initial amplitudes carry \emph{no information}.

        \vspace{4pt}
        \textbf{amp\_only} is the worst condition:
        good amplitudes with random biases performs
        worse than good biases with random amplitudes.

        \vspace{4pt}
        \textbf{Conclusion:}\\
        Bias position at $t=0$ is the sole
        informative quantity for collapse.
        The ``winning ticket'' is a good bias placement,
        not a good amplitude.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}
